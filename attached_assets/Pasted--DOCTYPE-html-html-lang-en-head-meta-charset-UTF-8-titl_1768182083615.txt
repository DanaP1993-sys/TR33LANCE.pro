  <script>
    const launchButton = document.getElementById('launchButton');

    launchButton.addEventListener('click', async () => {
      const password = document.getElementById('adminPassword').value;
      if (!password) return alert('Enter admin password!');

      try {
        // 1Ô∏è‚É£ Login to get JWT
        const loginRes = await fetch('https://tr33lance.pro/admin-login', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ password })
        });
        const loginData = await loginRes.json();
        if (!loginData.token) return alert('Invalid password!');
        const token = loginData.token;

        // 2Ô∏è‚É£ Send Launch Alert
        const alertRes = await fetch('https://tr33lance.pro/send-launch-alert', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'Authorization': 'Bearer ' + token
          }
        });
        const alertData = await alertRes.json();
        alert(`üöÄ Launch alert sent! Total devices: ${alertData.totalDevices}`);
      } catch (err) {
        console.error(err);
        alert('Failed to send launch alert!');
      }
    });
  </script>
